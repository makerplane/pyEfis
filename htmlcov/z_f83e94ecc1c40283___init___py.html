252 def processConditions(self,clicked=False): 

253 self._db_data['SCREEN'] = self.parent.screenName 

254 self._db_data['CLICKED'] = clicked 

255 self._db_data['DBKEY'] = self._dbkey.value 

256 self._db_data["PREVIOUS_CONDITION"] = False 

257 logger.debug(f"{self._dbkey.key}:{self._dbkey.value}") 

258 for cond in self._conditions: 

259 if 'when' in cond: 

260 if type(cond['when']) == str: 

261 expr = pc.to_struct(pc.tokenize(cond['when'], sep=' ', brkts='[]')) 

262 if pc.pycond(expr)(state=self._db_data) == True: 

263 self._db_data["PREVIOUS_CONDITION"] = True 

264 logger.debug(f"{self.parent.parent.getRunningScreen()}:{self._dbkey.key}:{cond['when']} = True") 

265 self.processActions(cond['actions']) 

266 logger.debug(f"{self.parent.parent.getRunningScreen()}:{self._dbkey.key}:{cond['when']} conditions processed") 

267 if not cond.get('continue', False): 

268 logger.debug(f"{self.parent.parent.getRunningScreen()}:{self._dbkey.key}:{cond['when']} Does not continue") 

269 return 

270 else: 

271 logger.debug(f"{self.parent.parent.getRunningScreen()}:{self._dbkey.key}:{cond['when']} continues") 

272 else: 

273 self._db_data["PREVIOUS_CONDITION"] = False 

274 logger.debug(f"{self.parent.parent.getRunningScreen()}:{self._dbkey.key}:{cond['when']} = False") 

275 elif type(cond['when']) == bool: 

276 if cond['when']: 

277 if self._button.isChecked() or self._toggle == False: 

278 self.processActions(cond['actions']) 

279 if not cond.get('continue', False): return 

280 else: 

281 if not self._button.isChecked(): 

282 self.processActions(cond['actions']) 

283 if not cond.get('continue', False): return
